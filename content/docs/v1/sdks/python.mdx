# Python SDK

The official Payfake SDK for Python.

## Installation

```bash
pip install payfake
```

## Quick Start

```python
from payfake import Client

client = Client(
    secret_key="sk_test_xxx",
    base_url="http://localhost:8080"
)

tx = client.transaction.initialize(
    email="customer@example.com",
    amount=10000,
    currency="GHS"
)

print(f"Authorization URL: {tx.authorization_url}")
print(f"Access Code: {tx.access_code}")
print(f"Reference: {tx.reference}")
```

## Authentication

```python
from payfake import Client

# Create auth client with empty secret key
auth_client = Client(
    secret_key="",
    base_url="http://localhost:8080"
)

# Register or login
try:
    reg = auth_client.auth.register(
        business_name="Acme Store",
        email="dev@acme.com",
        password="secret123"
    )
    token = reg.token
except PayfakeError as e:
    if e.is_code(PayfakeError.CODE_EMAIL_TAKEN):
        login = auth_client.auth.login(
            email="dev@acme.com",
            password="secret123"
        )
        token = login.token

# Get API keys
keys = auth_client.auth.get_keys(token)

# Create authenticated client
client = Client(
    secret_key=keys.secret_key,
    base_url="http://localhost:8080"
)
```

## Initialize Transaction

```python
from payfake.types import InitializeInput

tx = client.transaction.initialize(
    InitializeInput(
        email="customer@example.com",
        amount=10000,
        currency="GHS",
        reference="unique-ref-001",
        callback_url="https://yourapp.com/callback",
        channels=["card", "mobile_money"],
        metadata={"order_id": "12345"}
    )
)
```

## Charge Card

```python
from payfake.types import ChargeCardInput

try:
    charge = client.charge.card(
        ChargeCardInput(
            access_code=tx.access_code,
            card_number="4111111111111111",
            card_expiry="12/26",
            cvv="123",
            email="customer@example.com"
        )
    )
    print(f"Status: {charge.transaction.status}")
except PayfakeError as e:
    if e.is_code(PayfakeError.CODE_CHARGE_FAILED):
        print(f"Charge failed: {e.fields[0].message if e.fields else e.code}")
```

## Charge Mobile Money

```python
from payfake.types import ChargeMomoInput

momo = client.charge.mobile_money(
    ChargeMomoInput(
        access_code=tx.access_code,
        phone="+233241234567",
        provider="mtn",
        email="customer@example.com"
    )
)

# MoMo always returns pending immediately
print(f"Status: {momo.transaction.status}")  # "pending"
```

## Control Panel

```python
from payfake.types import UpdateScenarioInput, ForceTransactionInput, ListOptions

# Update scenario
scenario = auth_client.control.update_scenario(
    token,
    UpdateScenarioInput(
        failure_rate=0.5,
        delay_ms=2000
    )
)
print(f"Failure rate: {scenario.failure_rate}")

# Force transaction outcome
forced = auth_client.control.force_transaction(
    token,
    tx.reference,
    ForceTransactionInput(
        status="failed",
        error_code="CHARGE_INSUFFICIENT_FUNDS"
    )
)
print(f"Forced status: {forced.status}")

# Reset scenario
auth_client.control.reset_scenario(token)

# Get logs
logs = auth_client.control.get_logs(
    token,
    ListOptions(page=1, per_page=5)
)

for log in logs:
    print(f"{log.method} {log.path} -> {log.status_code}")
```

## Error Handling

```python
from payfake import PayfakeError

try:
    charge = client.charge.card(input)
except PayfakeError as e:
    if e.is_code(PayfakeError.CODE_CHARGE_FAILED):
        print("Charge failed")
        if e.fields:
            print(f"Reason: {e.fields[0].message}")
    elif e.is_code(PayfakeError.CODE_INVALID_CARD):
        print("Invalid card number")
    else:
        raise
```

## Available Error Codes

```python
PayfakeError.CODE_EMAIL_TAKEN          # AUTH_EMAIL_TAKEN
PayfakeError.CODE_INVALID_CREDENTIALS  # AUTH_INVALID_CREDENTIALS
PayfakeError.CODE_UNAUTHORIZED         # AUTH_UNAUTHORIZED
PayfakeError.CODE_CHARGE_FAILED        # CHARGE_FAILED
PayfakeError.CODE_INVALID_CARD         # CHARGE_INVALID_CARD
PayfakeError.CODE_CARD_EXPIRED         # CHARGE_CARD_EXPIRED
PayfakeError.CODE_INSUFFICIENT_FUNDS   # CHARGE_INSUFFICIENT_FUNDS
PayfakeError.CODE_MOMO_TIMEOUT         # CHARGE_MOMO_TIMEOUT
```

## Async Support

```python
import asyncio
from payfake import AsyncClient

async def main():
    client = AsyncClient(
        secret_key="sk_test_xxx",
        base_url="http://localhost:8080"
    )
    
    tx = await client.transaction.initialize(
        email="customer@example.com",
        amount=10000
    )
    
    print(tx.authorization_url)

asyncio.run(main())
```
