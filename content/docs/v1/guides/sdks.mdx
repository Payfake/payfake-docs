    <TabPanel value="go">
      ```go
      package main

      import (
          "context"
          "fmt"
          "log"
          payfake "github.com/payfake/payfake-go"
      )

      func main() {
          client := payfake.New(payfake.Config{
              SecretKey: "sk_test_xxx",
              BaseURL:   "http://localhost:8080",
          })

          tx, err := client.Transaction.Initialize(context.Background(), payfake.InitializeInput{
              Email:  "customer@example.com",
              Amount: 10000,
          })
          if err != nil {
              log.Fatal(err)
          }

          fmt.Println(tx.AuthorizationURL)
      }
      ```

          email="customer@example.com",
          amount=10000
      )

      ```

      use payfake::{Client, Config, InitializeInput};

      #[tokio::main]
      async fn main() -> Result<(), Box<dyn std::error::Error>> {
          let client = Client::new(Config {
              secret_key: "sk_test_xxx".to_string(),
              base_url: Some("http://localhost:8080".to_string()),
              timeout_secs: None,
          });

          let tx = client.transaction.initialize(InitializeInput {
              email: "customer@example.com".to_string(),
              amount: 10000,
              ..Default::default()
          }).await?;

          println!("{}", tx.authorization_url);
          Ok(())
      }
      ```

## Authentication Flow

All SDKs follow the same authentication pattern:

1. Create auth client with empty secret key
2. Register or login to get JWT token
3. Call `get_keys()` with token to retrieve actual API keys
4. Create authenticated client with real secret key

<Tabs defaultValue="javascript">
  <TabList>
    <Tab value="javascript">JavaScript</Tab>
    <Tab value="python">Python</Tab>
    <Tab value="go">Go</Tab>
  </TabList>
  <TabPanels>
    <TabPanel value="javascript">
      ```javascript
      // Step 1: Auth client
      const authClient = createClient({ secretKey: '', baseURL });

      // Step 2: Login
      const login = await authClient.auth.login({ email, password });

      // Step 3: Get keys
      const keys = await authClient.auth.getKeys(login.token);

      // Step 4: Real client
      const client = createClient({ secretKey: keys.secretKey, baseURL });
      ```
    </TabPanel>
    <TabPanel value="python">
      ```python
      # Step 1: Auth client
      auth_client = Client(secret_key="", base_url=base_url)

      # Step 2: Login
      login = auth_client.auth.login(email=email, password=password)

      # Step 3: Get keys
      keys = auth_client.auth.get_keys(login.token)

      # Step 4: Real client
      client = Client(secret_key=keys.secret_key, base_url=base_url)
      ```
    </TabPanel>
    <TabPanel value="go">
      ```go
      // Step 1: Auth client
      authClient := payfake.New(payfake.Config{SecretKey: "", BaseURL: baseURL})

      // Step 2: Login
      login, _ := authClient.Auth.Login(ctx, payfake.LoginInput{Email: email, Password: password})

      // Step 3: Get keys
      keys, _ := authClient.Auth.GetKeys(ctx, login.Token)

      // Step 4: Real client
      client := payfake.New(payfake.Config{SecretKey: keys.SecretKey, BaseURL: baseURL})
      ```
